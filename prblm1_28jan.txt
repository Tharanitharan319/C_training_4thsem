#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Structure for contact details
struct Contact {
    long long phone;
};

// Structure for bank customer
struct Customer {
    int accNo;
    struct Contact contact;
    struct Customer* next;
};

// Function to create a new customer node
struct Customer* createCustomer(int accNo, long long phone) {
    struct Customer* newCustomer = (struct Customer*)malloc(sizeof(struct Customer));
    newCustomer->accNo = accNo;
    newCustomer->contact.phone = phone;
    newCustomer->next = NULL;
    return newCustomer;
}

// Function to display all customers
void displayCustomers(struct Customer* head) {
    struct Customer* current = head;
    while (current != NULL) {
        printf("%d - %lld", current->accNo, current->contact.phone);
        if (current->next != NULL) {
            printf(" -> ");
        }
        current = current->next;
    }
    printf(" -> NULL\n");
}

int main() {
    // Create customer nodes manually
    struct Customer* customer1 = createCustomer(101, 9876543210);
    struct Customer* customer2 = createCustomer(102, 9123456789);
    
    // Link the nodes
    customer1->next = customer2;
    
    // Display the customer list
    printf("Bank Customer List:\n");
    displayCustomers(customer1);
    
    // Free memory
    free(customer1);
    free(customer2);
    
    return 0;
}
